    2. **`evals/redaction.py`** — canary for PROJECT.md constraint "API keys must never be persisted to campaign files, vaults, transcripts, checkpoints, or debug logs" and GAME_SPEC.md privacy-canary eval:
       ```python
       import re
       from dataclasses import dataclass, field

       # Patterns are DELIBERATELY shaped like common provider keys. They match anywhere
       # in a string. Any match in text that should be safe to display or persist is a
       # failure.
       DEFAULT_SECRET_PATTERNS: tuple[tuple[str, re.Pattern[str]], ...] = (
           ("openrouter_key",      re.compile(r"sk-or-v1-[A-Za-z0-9]{16,}")),
           ("openai_key",          re.compile(r"sk-[A-Za-z0-9]{20,}")),
           ("anthropic_key",       re.compile(r"sk-ant-[A-Za-z0-9\-_]{20,}")),
           ("bearer_header",       re.compile(r"(?i)authorization:\s*bearer\s+[A-Za-z0-9._\-]{16,}")),
           ("aws_access_key",      re.compile(r"AKIA[0-9A-Z]{16}")),
           ("high_entropy_hex",    re.compile(r"\b[a-f0-9]{48,}\b")),
       )

       @dataclass
       class RedactionHit:
           label: str
           match: str
           index: int

       @dataclass
       class RedactionCanary:
           patterns: tuple[tuple[str, re.Pattern[str]], ...] = DEFAULT_SECRET_PATTERNS

           def scan(self, text: str) -> list[RedactionHit]:
               hits: list[RedactionHit] = []
               for label, pat in self.patterns:
                   for m in pat.finditer(text):
                       hits.append(RedactionHit(label=label, match=m.group(0), index=m.start()))
               return hits
       ```
       No network, no provider import. Pure regex. This is the reusable object Phase 2's LLMClient/CostGovernor log redaction will assert against.

    3. **`evals/schema_round_trip.py`** — round-trip helpers:
       ```python
       import json
       from pathlib import Path
       from typing import TypeVar
       import pydantic

       from sagasmith.schemas.validation import validate_persisted_state, PersistedStateError

       M = TypeVar("M", bound=pydantic.BaseModel)

       def assert_round_trip(instance: M) -> None:
           data = instance.model_dump(mode="json")
           rebuilt = type(instance).model_validate(data)
           assert rebuilt == instance, (
               f"Round-trip mismatch for {type(instance).__name__}. "
               f"Original and rebuilt differ; inspect model_dump output."
           )

       def assert_fixture_round_trips(path: Path) -> None:
           data = json.loads(path.read_text(encoding="utf-8"))
           state = validate_persisted_state(data)
           roundtrip = validate_persisted_state(state.model_dump(mode="json"))
           assert roundtrip == state

       def assert_fixture_rejects(path: Path, expected_substring: str) -> None:
           data = json.loads(path.read_text(encoding="utf-8"))
           try:
               validate_persisted_state(data)
           except PersistedStateError as e:
               assert expected_substring in str(e), (
                   f"Expected {expected_substring!r} in rejection message; got {e!s}"
               )
               return
           raise AssertionError(f"Fixture {path.name} should have been rejected but validated")
       ```

  <action>
    Wrap Task 1's building blocks into a single developer entry point: `sagasmith smoke`. Two complementary execution modes:

    - `--mode fast` (default): runs the in-process `run_smoke()` harness. Fast (<1s), no pytest overhead, suitable for pre-commit and interactive feedback loops.
    - `--mode pytest`: shells out to `pytest -m smoke`. Catches everything a developer wrote into a test but not yet plumbed into the in-process harness. This is what `make smoke` already invokes.

    Both modes must exit 0 on success and a nonzero code on failure.

    1. **`evals/harness.py`**:
       ```python
       from __future__ import annotations
       import json
       import tempfile
       from dataclasses import dataclass, field
       from pathlib import Path

       from sagasmith.schemas.validation import validate_persisted_state, PersistedStateError
       from sagasmith.schemas.export import export_all_schemas, LLM_BOUNDARY_AND_PERSISTED_MODELS
       from sagasmith.evals.fixtures import make_valid_saga_state
       from sagasmith.evals.redaction import RedactionCanary
       from sagasmith.evals.schema_round_trip import assert_round_trip

       @dataclass
       class SmokeCheck:
           name: str
           ok: bool
           detail: str = ""

       @dataclass
       class SmokeResult:
           checks: list[SmokeCheck] = field(default_factory=list)

           @property
           def ok(self) -> bool:
               return all(c.ok for c in self.checks)

           def format(self) -> str:
               lines = [f"{'OK ' if c.ok else 'FAIL'} {c.name}" + (f" — {c.detail}" if c.detail else "") for c in self.checks]
               lines.append("")
               lines.append(f"{sum(c.ok for c in self.checks)}/{len(self.checks)} checks passed")
               return "\n".join(lines)

       def run_smoke() -> SmokeResult:
           result = SmokeResult()

           # Check 1: canonical SagaState round-trips.
           try:
               state = make_valid_saga_state()
               assert_round_trip(state)
               result.checks.append(SmokeCheck("schema.round_trip.saga_state", True))
           except Exception as e:  # noqa: BLE001
               result.checks.append(SmokeCheck("schema.round_trip.saga_state", False, str(e)))

           # Check 2: validate_persisted_state rejects a malformed payload.
           bad = state.model_dump(mode="json") if "state" in locals() else {}
           bad.pop("campaign_id", None)
           try:
               validate_persisted_state(bad)
               result.checks.append(SmokeCheck("schema.validation.rejects_missing_field", False,
                                               "Expected PersistedStateError; got none"))
           except PersistedStateError:
               result.checks.append(SmokeCheck("schema.validation.rejects_missing_field", True))
           except Exception as e:  # noqa: BLE001
               result.checks.append(SmokeCheck("schema.validation.rejects_missing_field", False,
                                               f"Wrong exception type: {type(e).__name__}"))

           # Check 3: JSON Schema export produces one file per declared model.
           try:
               with tempfile.TemporaryDirectory() as td:
                   paths = export_all_schemas(Path(td))
                   got = {p.stem.replace(".schema", "") for p in paths}
                   want = {m.__name__ for m in LLM_BOUNDARY_AND_PERSISTED_MODELS}
                   if got == want:
                       result.checks.append(SmokeCheck("schema.export.full_coverage", True))
                   else:
                       result.checks.append(SmokeCheck(
                           "schema.export.full_coverage", False,
                           f"missing={want - got} extra={got - want}",
                       ))
           except Exception as e:  # noqa: BLE001
               result.checks.append(SmokeCheck("schema.export.full_coverage", False, str(e)))

           # Check 4: Exported schemas contain no secret-shaped strings.
           try:
               with tempfile.TemporaryDirectory() as td:
                   paths = export_all_schemas(Path(td))
                   blob = "\n".join(p.read_text(encoding="utf-8") for p in paths)
                   hits = RedactionCanary().scan(blob)
                   if hits:
                       result.checks.append(SmokeCheck(
                           "redaction.exported_schemas_clean", False,
       ```

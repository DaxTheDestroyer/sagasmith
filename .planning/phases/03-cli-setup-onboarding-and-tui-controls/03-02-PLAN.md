```python
class OnboardingPhase(str, Enum):
    WELCOME = "welcome"
    TONE = "tone"
    PILLARS = "pillars"
    COMBAT_DICE_UX = "combat_dice_ux"
    CONTENT_POLICY = "content_policy"
    LENGTH_DEATH = "length_death"
    BUDGET = "budget"
    CHARACTER_MODE = "character_mode"
    REVIEW = "review"
    DONE = "done"

@dataclass(frozen=True)
class PhasePrompt:
    phase: OnboardingPhase
    title: str
    description: str
    fields: list[PromptField]   # each field has id, kind, choices, validation hint

@dataclass(frozen=True)
class StepResult:
    next_prompt: PhasePrompt | None  # None when phase == DONE
    errors: list[str]                 # non-fatal; wizard stays on same phase if non-empty
    phase: OnboardingPhase            # current phase after step()

class OnboardingWizard:
    def current_prompt(self) -> PhasePrompt: ...
    def step(self, answers: dict[str, object]) -> StepResult: ...
    def review(self) -> dict[str, object]: ...        # dict-of-dicts for read-only inspection
    def edit(self, field_path: str, value: object) -> list[str]: ...   # e.g., "profile.pacing" or "content_policy.hard_limits"
    def build_records(self) -> tuple[PlayerProfile, ContentPolicy, HouseRules]: ...
    @property
    def is_complete(self) -> bool: ...
```

Define:
```python
class PromptFieldKind(str, Enum):
    FREE_TEXT = "free_text"             # one-line string
    MULTI_TEXT = "multi_text"           # list of strings (comma-separated or list input)
    SINGLE_CHOICE = "single_choice"     # pick one of choices
    MULTI_CHOICE = "multi_choice"       # pick zero-or-more of choices
    PILLAR_BUDGET = "pillar_budget"     # four ints that sum to 10 → normalized to weights
    SOFT_LIMIT_MAP = "soft_limit_map"   # dict of topic → {fade_to_black|avoid_detail|ask_first}
    FLOAT = "float"                     # non-negative float
    BOOL = "bool"
```

```python
@dataclass(frozen=True)
class PromptField:
    id: str
    label: str
    kind: PromptFieldKind
    choices: tuple[str, ...] = ()
    required: bool = True
    help_text: str = ""

@dataclass(frozen=True)
class PhasePrompt:
    phase: OnboardingPhase
    title: str
    description: str
    fields: tuple[PromptField, ...]
```

Build `ONBOARDING_PHASES: tuple[PhasePrompt, ...]` in this order (matches GAME_SPEC §7.1):

- **WELCOME**: title "Welcome + premise", fields: `genre` (MULTI_TEXT, help "Name 1–3 genres, e.g. high_fantasy, cosmic_horror").
- **TONE**: fields: `tone` (MULTI_TEXT, help "1–5 tone keywords"), `touchstones` (MULTI_TEXT, help "Name 3 books/games/films the story should feel like").
- **PILLARS**: fields: `pillar_budget` (PILLAR_BUDGET with 4 labeled subkeys: combat, exploration, social, puzzle — sums must equal 10 int), `pacing` (SINGLE_CHOICE, choices=("slow","medium","fast")).
- **COMBAT_DICE_UX**: fields: `combat_style` (SINGLE_CHOICE, choices=("theater_of_mind",), required=True — MVP only offers one), `dice_ux` (SINGLE_CHOICE, choices=("auto","reveal","hidden")).
- **CONTENT_POLICY**: fields: `hard_limits` (MULTI_TEXT, required=False, help "Topics to redline entirely"), `soft_limits` (SOFT_LIMIT_MAP, required=False), `preferences` (MULTI_TEXT, required=False).
- **LENGTH_DEATH**: fields: `campaign_length` (SINGLE_CHOICE, choices=("one_shot","arc","open_ended")), `death_policy` (SINGLE_CHOICE, choices=("hardcore","heroic_recovery","retire_and_continue")).
- **BUDGET**: fields: `per_session_usd` (FLOAT, help "Per-session USD cap, e.g. 2.50"), `hard_stop` (BOOL, default True).
- **CHARACTER_MODE**: fields: `character_mode` (SINGLE_CHOICE, choices=("guided","player_led","pregenerated")).
- **REVIEW**: fields: `review_confirmed` (BOOL). No new data gathered — user either commits or edits.

Define `PHASE_ORDER: tuple[OnboardingPhase, ...]` in the above sequence plus `DONE` as terminal.

Also define `parse_answer(field: PromptField, raw: object) -> tuple[object, list[str]]` with one branch per `PromptFieldKind`:
- FREE_TEXT: accept str; strip; if required and empty → error.
- MULTI_TEXT: accept `list[str]` or single str (comma-split). Strip blank items. If required and empty → error.
- SINGLE_CHOICE: must be in `choices`; else error `"must be one of {choices}"`.
- MULTI_CHOICE: subset of `choices`.
- PILLAR_BUDGET: accept `dict[str, int]` with exactly the 4 required keys; each ≥ 0; sum == 10. Returns normalized `dict[str, float]` (each / 10.0). If sum != 10 → `"pillar points must sum to 10"`.
- SOFT_LIMIT_MAP: accept `dict[str, str]` where each value ∈ {"fade_to_black","avoid_detail","ask_first"}. Keys must be non-empty. Empty map OK.
- FLOAT: accept int/float/str; convert; require ≥ 0.
- BOOL: accept bool or {"true","false","yes","no","y","n","1","0"}.

Return both the parsed value and a list of error strings (empty on success). The wizard accumulates errors per step.

```python
@dataclass
class _DraftState:
    # Per-phase raw answer slots; filled incrementally.
    genre: list[str] | None = None
    tone: list[str] | None = None
    touchstones: list[str] | None = None
    pillar_weights: dict[str, float] | None = None  # normalized
    pacing: str | None = None
    combat_style: str | None = None
    dice_ux: str | None = None
    hard_limits: list[str] | None = None
    soft_limits: dict[str, str] | None = None
    preferences: list[str] | None = None
    campaign_length: str | None = None
    death_policy: str | None = None
    per_session_usd: float | None = None
    hard_stop: bool | None = None
    character_mode: str | None = None
```

`class OnboardingWizard`:

- `__init__(self) -> None`: `_phase = OnboardingPhase.WELCOME`, `_draft = _DraftState()`.
- `current_prompt()`: return `ONBOARDING_PHASES[PHASE_ORDER.index(self._phase)]` (except when phase == DONE, raise `RuntimeError("wizard complete")`).
- `step(answers: dict[str, object]) -> StepResult`:
  - For the current prompt, `parse_answer` each declared field. Collect errors.
  - If any errors, return `StepResult(next_prompt=current_prompt(), errors=errors, phase=self._phase)` — DO NOT advance.
  - Write parsed values into `_draft` by field name.
  - Advance `_phase` to the next in PHASE_ORDER.
  - If the new phase is REVIEW: return `StepResult(next_prompt=current_prompt(), errors=[], phase=REVIEW)`.
  - If the new phase is DONE: return `StepResult(next_prompt=None, errors=[], phase=DONE)`. (Caller has already confirmed via `review_confirmed=True` step.)
- `review() -> dict[str, object]`: return a nested dict:
  ```python
  {
    "profile": PlayerProfile(...).model_dump(),   # raise ValidationError upward if _draft incomplete
    "content_policy": ContentPolicy(...).model_dump(),
    "house_rules": HouseRules(...).model_dump(),
  }
  ```
  If any required draft field is None → raise `RuntimeError("review called before all phases filled")`.
- `edit(field_path: str, value: object) -> list[str]`:
  - `field_path` is dotted: `"profile.pacing"`, `"content_policy.hard_limits"`, `"house_rules.dice_ux"`, `"profile.budget.per_session_usd"`.
  - Apply the edit to `_draft` fields. Re-validate only the affected record by constructing a fresh Pydantic model from the current draft.
  - Return list of validation error strings (empty on success).
  - Disallow edits to `profile.combat_style` (Literal["theater_of_mind"] — fixed in MVP per STATE_SCHEMA §4.1); return `["combat_style is fixed to 'theater_of_mind' in MVP"]`.
- `build_records() -> tuple[PlayerProfile, ContentPolicy, HouseRules]`:
  - Construct the three models from `_draft`. `PlayerProfile.pillar_weights` uses normalized `_draft.pillar_weights`. `PlayerProfile.budget = BudgetPolicy(per_session_usd=..., hard_stop=...)`.
  - For `HouseRules`: `dice_ux` mirrors `_draft.dice_ux`; MVP defaults: `initiative_visible=True`, `allow_retcon=True`, `auto_save_every_turn=True`, `session_end_trigger="player_command_or_budget"`.
  - Returns the triple.
- `is_complete` property: `self._phase == OnboardingPhase.DONE`.

Commit check on `step()` when phase == REVIEW: if `answers["review_confirmed"]` is truthy → advance to DONE. Otherwise stay on REVIEW and return errors=["review not confirmed"]. This gives the caller explicit control (ONBD-03).

2. **`src/sagasmith/onboarding/store.py`**:

```python
@dataclass(frozen=True)
class OnboardingTriple:
    player_profile: PlayerProfile
    content_policy: ContentPolicy
    house_rules: HouseRules

@dataclass(frozen=True)
class OnboardingStore:
    conn: sqlite3.Connection

    def commit(self, campaign_id: str, triple: OnboardingTriple) -> None:
        """Write all three records atomically. INSERT OR REPLACE so re-run works."""
        now = datetime.now(UTC).isoformat()
        with self.conn:     # atomic transaction
            self.conn.execute(
                "INSERT OR REPLACE INTO onboarding_player_profile (campaign_id, profile_json, committed_at) VALUES (?, ?, ?)",
                (campaign_id, triple.player_profile.model_dump_json(), now),
            )
            self.conn.execute(
                "INSERT OR REPLACE INTO onboarding_content_policy (campaign_id, policy_json, committed_at) VALUES (?, ?, ?)",
                (campaign_id, triple.content_policy.model_dump_json(), now),
            )
            self.conn.execute(
                "INSERT OR REPLACE INTO onboarding_house_rules (campaign_id, rules_json, committed_at) VALUES (?, ?, ?)",
                (campaign_id, triple.house_rules.model_dump_json(), now),
            )

    def reload(self, campaign_id: str) -> OnboardingTriple | None:
        """Return the committed triple, or None if any row missing.

        Partial state (one table populated, another not) raises TrustServiceError —
        means a commit was interrupted or the DB was hand-edited.
        """

    def exists(self, campaign_id: str) -> bool:
        """True if all three rows present for this campaign."""
```

`reload` behavior:
- Query all three tables for `campaign_id`.
- If all three missing → return None (fresh campaign, no onboarding yet).
- If all three present → `PlayerProfile.model_validate_json(...)`, etc., return `OnboardingTriple`.
- If 1 or 2 present (partial) → raise `TrustServiceError(f"partial onboarding state for {campaign_id}: missing {list_of_missing_tables}")`.

Import `TrustServiceError` from `sagasmith.services.errors`. Import `PlayerProfile`, `ContentPolicy`, `HouseRules` from `sagasmith.schemas.player`. Import `datetime, UTC` from stdlib.

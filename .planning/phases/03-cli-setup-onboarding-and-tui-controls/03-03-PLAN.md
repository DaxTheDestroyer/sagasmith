```python
@dataclass(frozen=True)
class StatusSnapshot:
    hp_current: int | None = None
    hp_max: int | None = None
    conditions: tuple[str, ...] = ()
    active_quest: str | None = None
    location: str | None = None
    clock_day: int | None = None
    clock_hhmm: str | None = None
    last_rolls: tuple[str, ...] = ()

    def format_hp(self) -> str:
        if self.hp_current is None or self.hp_max is None:
            return "HP: —"
        return f"HP: {self.hp_current}/{self.hp_max}"

    def format_clock(self) -> str:
        if self.clock_day is None or self.clock_hhmm is None:
            return "Clock: —"
        return f"Day {self.clock_day}, {self.clock_hhmm}"

@dataclass
class TUIState:
    """Mutable UI state — NOT the graph SagaState. Plan 04 will bridge."""
    status: StatusSnapshot = field(default_factory=StatusSnapshot)
    scrollback: list[str] = field(default_factory=list)   # rendered transcript lines
```

```python
class NarrationArea(Widget):
    DEFAULT_CSS = """
    NarrationArea { height: 1fr; border: solid $primary; }
    """

    def compose(self):
        yield RichLog(id="narration-log", wrap=True, highlight=False, markup=False, auto_scroll=True)

    def append_line(self, text: str) -> None:
        log = self.query_one("#narration-log", RichLog)
        log.write(text)

    def load_scrollback(self, lines: list[str]) -> None:
        log = self.query_one("#narration-log", RichLog)
        for line in lines:
            log.write(line)
```

```python
from textual.widget import Widget
from textual.widgets import Static
from textual.reactive import reactive
from sagasmith.tui.state import StatusSnapshot

class StatusPanel(Widget):
    DEFAULT_CSS = """
    StatusPanel { width: 32; border: solid $accent; padding: 1; }
    """

    snapshot: reactive[StatusSnapshot] = reactive(StatusSnapshot(), always_update=True)

    def compose(self):
        yield Static(id="status-body")

    def watch_snapshot(self, new: StatusSnapshot) -> None:
        body = self.query_one("#status-body", Static)
        body.update(self._render(new))

    def _render(self, s: StatusSnapshot) -> str:
        lines = [
            s.format_hp(),
            "Conditions: " + (", ".join(s.conditions) if s.conditions else "—"),
            f"Quest: {s.active_quest or '—'}",
            f"Location: {s.location or '—'}",
            s.format_clock(),
            "Last rolls:",
            *[f"  {r}" for r in (s.last_rolls or ("—",))[:3]],
        ]
        return "\n".join(lines)
```

```python
from pathlib import Path
from textual.app import App, ComposeResult
from textual.containers import Horizontal, Vertical
from textual.message import Message

from sagasmith.app.campaign import open_campaign
from sagasmith.app.paths import CampaignPaths
from sagasmith.schemas.campaign import CampaignManifest
from sagasmith.tui.state import StatusSnapshot, TUIState
from sagasmith.tui.widgets.narration import NarrationArea
from sagasmith.tui.widgets.status_panel import StatusPanel
from sagasmith.tui.widgets.safety_bar import SafetyBar
from sagasmith.tui.widgets.input_line import InputLine

class PlayerInputSubmitted(Message):
    def __init__(self, text: str) -> None:
        self.text = text
        super().__init__()

class CommandInvoked(Message):
    def __init__(self, raw: str, name: str, args: tuple[str, ...]) -> None:
        self.raw = raw
        self.name = name
        self.args = args
        super().__init__()

class SagaSmithApp(App):
    CSS = """
    #main-row { height: 1fr; }
    """
    BINDINGS = [
        ("ctrl+q", "quit", "Quit"),
    ]

    def __init__(self, paths: CampaignPaths, manifest: CampaignManifest) -> None:
        super().__init__()
        self.paths = paths
        self.manifest = manifest
        self.state = TUIState()
        # registry is set by runtime.build_app BEFORE mount (see task 2).
        self.commands = None    # type: CommandRegistry | None

    def compose(self) -> ComposeResult:
        yield SafetyBar()
        with Horizontal(id="main-row"):
            yield NarrationArea(id="narration-area")
            yield StatusPanel(id="status-panel")
        yield InputLine(id="input-line")

    def on_mount(self) -> None:
        narration = self.query_one(NarrationArea)
        narration.append_line(f"Campaign: {self.manifest.campaign_name}")
        narration.append_line("Type /help for commands, or describe what your character does.")
        status = self.query_one(StatusPanel)
        status.snapshot = self.state.status     # triggers watcher

    def on_input_line_submitted(self, event: InputLine.Submitted) -> None:
        raw = event.raw
        if raw.startswith("/"):
            # parse slash command
            body = raw[1:]
            tokens = body.split()
            if not tokens:
                return
            name, *args = tokens
            self.post_message(CommandInvoked(raw=raw, name=name, args=tuple(args)))
        else:
            self.post_message(PlayerInputSubmitted(text=raw))

    def on_player_input_submitted(self, event: PlayerInputSubmitted) -> None:
        narration = self.query_one(NarrationArea)
        narration.append_line(f"> {event.text}")
        self.state.scrollback.append(f"> {event.text}")

    def on_command_invoked(self, event: CommandInvoked) -> None:
        if self.commands is None:
            # During Plan 03-03 only /help is wired in on_mount via runtime.
            # Plan 03-04 will replace this with full registry dispatch.
            return
        self.commands.dispatch(self, event.name, event.args)
```

1. **`src/sagasmith/tui/commands/registry.py`**:

```python
from __future__ import annotations
from typing import TYPE_CHECKING, Protocol, runtime_checkable

if TYPE_CHECKING:
    from sagasmith.tui.app import SagaSmithApp

@runtime_checkable
class TUICommand(Protocol):
    name: str
    description: str
    def handle(self, app: "SagaSmithApp", args: tuple[str, ...]) -> None: ...

class CommandRegistry:
    def __init__(self) -> None:
        self._by_name: dict[str, TUICommand] = {}

    def register(self, command: TUICommand) -> None:
        if command.name in self._by_name:
            raise ValueError(f"duplicate command: {command.name}")
        self._by_name[command.name] = command

    def get(self, name: str) -> TUICommand | None:
        return self._by_name.get(name)

    def names(self) -> list[str]:
        return sorted(self._by_name.keys())

    def all(self) -> list[TUICommand]:
        return [self._by_name[n] for n in self.names()]

    def dispatch(self, app: "SagaSmithApp", name: str, args: tuple[str, ...]) -> None:
        command = self.get(name)
        if command is None:
            narration = app.query_one("NarrationArea")
            narration.append_line(f"Unknown command: /{name}. Type /help for a list.")
            return
        command.handle(app, args)
```

```python
def build_app(campaign_root: Path) -> SagaSmithApp:

    app.commands = registry

    # Load recent transcript for scrollback (TUI-03).
    app.initial_scrollback = _load_scrollback(paths.db)
    return app

def _load_scrollback(db_path: Path) -> list[str]:
    """Return last SCROLLBACK_LIMIT rendered lines from transcript_entries.

    Rendering rule:
      - kind='player_input'   → f"> {content}"
      - kind='narration_final'→ content (verbatim)
      - kind='system_note'    → f"[{content}]"
    """
    lines: list[str] = []
    conn = open_campaign_db(db_path, read_only=True)
    try:
        rows = conn.execute(
            """
            SELECT kind, content
              FROM transcript_entries
             ORDER BY id DESC
             LIMIT ?
            """,
            (SCROLLBACK_LIMIT,),
        ).fetchall()
    finally:
        conn.close()
    # Rows are newest-first; reverse for chronological playback.
    for kind, content in reversed(rows):
        if kind == "player_input":
            lines.append(f"> {content}")
        elif kind == "narration_final":
            lines.append(content)
        else:
            lines.append(f"[{content}]")
    return lines
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User input line → Textual Input event → app message routing | Free-text input is untrusted; markup is disabled on the narration log (`markup=False`) to prevent Rich escape-sequence injection via transcripts. |
| Scrollback load → SQLite transcript_entries | The app reads persisted transcript verbatim; with `markup=False` the RichLog cannot interpret `[red]...[/red]`-style content as formatting, eliminating a stored-XSS-equivalent injection surface. |
| CommandRegistry.dispatch → command.handle | Commands are trusted code registered at runtime startup; only first-party modules register. Plan 03-04's commands run with full app access. |
| sagasmith play CLI → build_app → Textual App | `open_campaign` validation runs before app construction; malformed campaign dir exits with code 2 before any UI mount. |

|-----------|----------|-----------|-------------|-----------------|
